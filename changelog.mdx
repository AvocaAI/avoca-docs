## All Recent Updates

<Tabs>
  <Tab title="By Date">
    {changelogData.releases.map((release, index) => (
      <Update key={index} label={release.date}>
        <h3>{release.product}</h3>
        {release.features.map((feature, fIndex) => (
          <div key={fIndex}>
            <h4>{feature.title}</h4>
            {feature.author && <p><em>Author: {feature.author}</em></p>}
            {feature.description && <p>{feature.description}</p>}
            <ul>
              {feature.items.map((item, iIndex) => (
                <li key={iIndex}>
                  <strong>{item.label}</strong> - {item.description}
                </li>
              ))}
            </ul>
            {feature.note && <Note>{feature.note}</Note>}
          </div>
        ))}
      </Update>
    ))}
  </Tab>
  <Tab title="By Product">
    {Object.entries(
      changelogData.releases.reduce((acc, release) => {
        if (!acc[release.product]) acc[release.product] = [];
        acc[release.product].push(release);
        return acc;
      }, {})
    ).map(([product, releases]) => (
      <div key={product}>
        <h3>{product}</h3>
        {releases.map((release, rIndex) => (
          <Update key={rIndex} label={release.date}>
            {release.features.map((feature, fIndex) => (
              <div key={fIndex}>
                <h4>{feature.title}</h4>
                {feature.author && <p><em>Author: {feature.author}</em></p>}
                {feature.description && <p>{feature.description}</p>}
                <ul>
                  {feature.items.map((item, iIndex) => (
                    <li key={iIndex}>
                      <strong>{item.label}</strong> - {item.description}
                    </li>
                  ))}
                </ul>
                {feature.note && <Note>{feature.note}</Note>}
              </div>
            ))}
          </Update>
        ))}
      </div>
    ))}
  </Tab>
</Tabs>
