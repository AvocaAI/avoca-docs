## All Recent Updates

{(() => {
  // Filter for past updates only (using 6 AM ET cutoff)
  const now = new Date();
  
  // Get today's 6 AM ET as the cutoff
  const cutoff = new Date();
  const etOffset = -5; // ET is UTC-5 (or UTC-4 during daylight saving)
  cutoff.setUTCHours(6 - etOffset, 0, 0, 0); // 6 AM ET = 11 AM UTC (standard) or 10 AM UTC (daylight)
  
  // If we're before 6 AM ET, use yesterday's date
  if (now < cutoff) {
    cutoff.setDate(cutoff.getDate() - 1);
  }
  
  const cutoffISO = cutoff.toISOString().split('T')[0];
  
  const pastUpdates = updatesData.updates.filter(update => {
    return update.date <= cutoffISO;
  });
  
  // Group by date for display
  const groupedByDate = pastUpdates.reduce((acc, update) => {
    if (!acc[update.displayDate]) acc[update.displayDate] = [];
    acc[update.displayDate].push(update);
    return acc;
  }, {});
  
  // Sort dates in descending order (most recent first)
  const sortedDates = Object.keys(groupedByDate).sort((a, b) => {
    const dateA = pastUpdates.find(u => u.displayDate === a).date;
    const dateB = pastUpdates.find(u => u.displayDate === b).date;
    return dateB.localeCompare(dateA);
  });
  
  if (sortedDates.length === 0) {
    return <p>No completed updates yet. Check our <a href="/roadmap">Roadmap</a> for upcoming features!</p>;
  }
  
  return (
    <Tabs>
      <Tab title="By Date">
        {sortedDates.map(date => (
          <Update key={date} label={date}>
            {groupedByDate[date].map((feature, fIndex) => (
              <div key={fIndex} style={{ marginBottom: '2rem' }}>
                <h3>{feature.product} - {feature.title}</h3>
                {feature.author && <p><em>Author: {feature.author}</em></p>}
                {feature.description && (
                  <>
                    <br />
                    <p>{feature.description}</p>
                  </>
                )}
                {feature.loomUrl && (
                  <div style={{ position: 'relative', paddingBottom: '56.25%', height: 0, overflow: 'hidden', margin: '1rem 0' }}>
                    <iframe 
                      src={`https://www.loom.com/embed/${feature.loomUrl.split('/').pop()}?hideEmbedTopBar=true&hide_share=true&hide_title=true&hide_owner=true&hide_speed=true`}
                      frameBorder="0"
                      webkitallowfullscreen 
                      mozallowfullscreen 
                      allowFullScreen
                      style={{ position: 'absolute', top: 0, left: 0, width: '100%', height: '100%', borderRadius: '8px' }}
                    />
                  </div>
                )}
                {feature.items && feature.items.length > 0 && (
                  <ul>
                    {feature.items.map((item, iIndex) => (
                      <li key={iIndex}>
                        <strong>{item.label}</strong> - {item.description}
                      </li>
                    ))}
                  </ul>
                )}
                {feature.note && <Note>{feature.note}</Note>}
              </div>
            ))}
          </Update>
        ))}
      </Tab>
      <Tab title="By Product">
        {Object.entries(
          pastUpdates.reduce((acc, update) => {
            if (!acc[update.product]) acc[update.product] = [];
            acc[update.product].push(update);
            return acc;
          }, {})
        ).sort().map(([product, updates]) => (
          <div key={product}>
            <h3>{product}</h3>
            {Object.entries(
              updates.reduce((acc, update) => {
                if (!acc[update.displayDate]) acc[update.displayDate] = [];
                acc[update.displayDate].push(update);
                return acc;
              }, {})
            ).sort((a, b) => {
              const dateA = updates.find(u => u.displayDate === a[0]).date;
              const dateB = updates.find(u => u.displayDate === b[0]).date;
              return dateB.localeCompare(dateA);
            }).map(([date, dateUpdates]) => (
              <Update key={date} label={date}>
                {dateUpdates.map((feature, fIndex) => (
                  <div key={fIndex} style={{ marginBottom: '1.5rem' }}>
                    <h4>{feature.title}</h4>
                    {feature.author && <p><em>Author: {feature.author}</em></p>}
                    {feature.description && <p>{feature.description}</p>}
                    {feature.loomUrl && (
                      <div style={{ margin: '1rem 0' }}>
                        <iframe 
                          src={`https://www.loom.com/embed/${feature.loomUrl.split('/').pop()}`}
                          frameBorder="0"
                          webkitallowfullscreen 
                          mozallowfullscreen 
                          allowFullScreen
                          style={{ width: '100%', height: '400px', borderRadius: '8px' }}
                        />
                      </div>
                    )}
                    {feature.items && feature.items.length > 0 && (
                      <ul>
                        {feature.items.map((item, iIndex) => (
                          <li key={iIndex}>
                            <strong>{item.label}</strong> - {item.description}
                          </li>
                        ))}
                      </ul>
                    )}
                    {feature.note && <Note>{feature.note}</Note>}
                  </div>
                ))}
              </Update>
            ))}
          </div>
        ))}
      </Tab>
    </Tabs>
  );
})()}
