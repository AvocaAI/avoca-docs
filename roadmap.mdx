## All Upcoming Features

<Tabs>
  <Tab title="By Date">
    {roadmapData.updates.map((update, index) => (
      <Update key={index} label={update.date}>
        <h3>{update.product}</h3>
        {update.features.map((feature, fIndex) => (
          <div key={fIndex}>
            <h4>{feature.title}</h4>
            {feature.author && <p><em>Author: {feature.author}</em></p>}
            {feature.description && <p>{feature.description}</p>}
            <ul>
              {feature.items.map((item, iIndex) => (
                <li key={iIndex}>
                  <strong>{item.label}</strong> - {item.description}
                </li>
              ))}
            </ul>
            {feature.note && <Note>{feature.note}</Note>}
          </div>
        ))}
      </Update>
    ))}
  </Tab>
  <Tab title="By Product">
    {Object.entries(
      roadmapData.updates.reduce((acc, update) => {
        if (!acc[update.product]) acc[update.product] = [];
        acc[update.product].push(update);
        return acc;
      }, {})
    ).map(([product, updates]) => (
      <div key={product}>
        <h3>{product}</h3>
        {updates.map((update, uIndex) => (
          <Update key={uIndex} label={update.date}>
            {update.features.map((feature, fIndex) => (
              <div key={fIndex}>
                <h4>{feature.title}</h4>
                {feature.author && <p><em>Author: {feature.author}</em></p>}
                {feature.description && <p>{feature.description}</p>}
                <ul>
                  {feature.items.map((item, iIndex) => (
                    <li key={iIndex}>
                      <strong>{item.label}</strong> - {item.description}
                    </li>
                  ))}
                </ul>
                {feature.note && <Note>{feature.note}</Note>}
              </div>
            ))}
          </Update>
        ))}
      </div>
    ))}
  </Tab>
</Tabs>
