## All Upcoming Features

{(() => {
  // Filter for current and future updates (using 6 AM ET cutoff)
  const now = new Date();
  
  // Get today's 6 AM ET as the cutoff
  const cutoff = new Date();
  const etOffset = -5; // ET is UTC-5 (or UTC-4 during daylight saving)
  cutoff.setUTCHours(6 - etOffset, 0, 0, 0); // 6 AM ET = 11 AM UTC (standard) or 10 AM UTC (daylight)
  
  // If we're before 6 AM ET, use yesterday's date
  if (now < cutoff) {
    cutoff.setDate(cutoff.getDate() - 1);
  }
  
  const cutoffISO = cutoff.toISOString().split('T')[0];
  
  const futureUpdates = updatesData.updates.filter(update => {
    return update.date > cutoffISO;
  });
  
  // Group by date for display
  const groupedByDate = futureUpdates.reduce((acc, update) => {
    if (!acc[update.displayDate]) acc[update.displayDate] = [];
    acc[update.displayDate].push(update);
    return acc;
  }, {});
  
  // Sort dates in ascending order (earliest first)
  const sortedDates = Object.keys(groupedByDate).sort((a, b) => {
    const dateA = futureUpdates.find(u => u.displayDate === a).date;
    const dateB = futureUpdates.find(u => u.displayDate === b).date;
    return dateA.localeCompare(dateB);
  });
  
  if (sortedDates.length === 0) {
    return <p>No upcoming features scheduled at this time. Check back soon for updates!</p>;
  }
  
  return (
    <Tabs>
      <Tab title="By Date">
        {sortedDates.map(date => (
          <Update key={date} label={date}>
            {Object.entries(
              groupedByDate[date].reduce((acc, update) => {
                if (!acc[update.product]) acc[update.product] = [];
                acc[update.product].push(update);
                return acc;
              }, {})
            ).map(([product, features]) => (
              <div key={product}>
                <h3>{product}</h3>
                {features.map((feature, fIndex) => (
                  <div key={fIndex}>
                    <h4>{feature.title}</h4>
                    {feature.author && <p><em>Author: {feature.author}</em></p>}
                    {feature.description && (
                      <>
                        <br />
                        <p>{feature.description}</p>
                      </>
                    )}
                    {feature.loomUrl && (
                      <div style={{ position: 'relative', paddingBottom: '56.25%', height: 0, overflow: 'hidden', margin: '1rem 0' }}>
                        <iframe 
                          src={`https://www.loom.com/embed/${feature.loomUrl.split('/').pop()}?hideEmbedTopBar=true&hide_share=true&hide_title=true&hide_owner=true&hide_speed=true`}
                          frameBorder="0"
                          webkitallowfullscreen 
                          mozallowfullscreen 
                          allowFullScreen
                          style={{ position: 'absolute', top: 0, left: 0, width: '100%', height: '100%', borderRadius: '8px' }}
                        />
                      </div>
                    )}
                    {feature.items && feature.items.length > 0 && (
                      <ul>
                        {feature.items.map((item, iIndex) => (
                          <li key={iIndex}>
                            <strong>{item.label}</strong> - {item.description}
                          </li>
                        ))}
                      </ul>
                    )}
                    {feature.note && <Note>{feature.note}</Note>}
                  </div>
                ))}
              </div>
            ))}
          </Update>
        ))}
      </Tab>
      <Tab title="By Product">
        {Object.entries(
          futureUpdates.reduce((acc, update) => {
            if (!acc[update.product]) acc[update.product] = [];
            acc[update.product].push(update);
            return acc;
          }, {})
        ).sort((a, b) => {
          // Sort products by their earliest date
          const earliestDateA = Math.min(...a[1].map(f => new Date(f.date).getTime()));
          const earliestDateB = Math.min(...b[1].map(f => new Date(f.date).getTime()));
          return earliestDateA - earliestDateB;
        }).map(([product, features]) => (
          <div key={product}>
            <h3>{product}</h3>
            {Object.entries(
              features.reduce((acc, feature) => {
                if (!acc[feature.displayDate]) acc[feature.displayDate] = [];
                acc[feature.displayDate].push(feature);
                return acc;
              }, {})
            ).sort((a, b) => {
              const dateA = features.find(f => f.displayDate === a[0]).date;
              const dateB = features.find(f => f.displayDate === b[0]).date;
              return dateA.localeCompare(dateB);
            }).map(([date, dateFeatures]) => (
              <Update key={date} label={date}>
                {dateFeatures.map((feature, fIndex) => (
                  <div key={fIndex}>
                    <h4>{feature.title}</h4>
                    {feature.author && <p><em>Author: {feature.author}</em></p>}
                    {feature.description && (
                      <>
                        <br />
                        <p>{feature.description}</p>
                      </>
                    )}
                    {feature.loomUrl && (
                      <div style={{ position: 'relative', paddingBottom: '56.25%', height: 0, overflow: 'hidden', margin: '1rem 0' }}>
                        <iframe 
                          src={`https://www.loom.com/embed/${feature.loomUrl.split('/').pop()}?hideEmbedTopBar=true&hide_share=true&hide_title=true&hide_owner=true&hide_speed=true`}
                          frameBorder="0"
                          webkitallowfullscreen 
                          mozallowfullscreen 
                          allowFullScreen
                          style={{ position: 'absolute', top: 0, left: 0, width: '100%', height: '100%', borderRadius: '8px' }}
                        />
                      </div>
                    )}
                    {feature.items && feature.items.length > 0 && (
                      <ul>
                        {feature.items.map((item, iIndex) => (
                          <li key={iIndex}>
                            <strong>{item.label}</strong> - {item.description}
                          </li>
                        ))}
                      </ul>
                    )}
                    {feature.note && <Note>{feature.note}</Note>}
                  </div>
                ))}
              </Update>
            ))}
          </div>
        ))}
      </Tab>
    </Tabs>
  );
})()}
