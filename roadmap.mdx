## All Upcoming Features

<Tabs>
  <Tab title="By Date">
    {Object.entries(
      roadmapData.updates.reduce((acc, update) => {
        if (!acc[update.date]) acc[update.date] = [];
        acc[update.date].push(update);
        return acc;
      }, {})
    ).map(([date, updates]) => (
      <Update key={date} label={date}>
        {Object.entries(
          updates.reduce((acc, update) => {

            acc[update.product].push(update);
            return acc;
          }, {})
        ).map(([product, features]) => (
          <div key={product}>
            <h3>{product}</h3>
            {features.map((feature, fIndex) => (
              <div key={fIndex}>
                <h4>{feature.title}</h4>
                {feature.author && <p><em>Author: {feature.author}</em></p>}
                {feature.description && (
                  <>
                    <br />
                    <p>{feature.description}</p>
                  </>
                )}
                {feature.items && feature.items.length > 0 && (
                  <ul>
                    {feature.items.map((item, iIndex) => (
                      <li key={iIndex}>
                        <strong>{item.label}</strong> - {item.description}
                      </li>
                    ))}
                  </ul>
                )}
                {feature.note && <Note>{feature.note}</Note>}
              </div>
            ))}
          </div>
        ))}
      </Update>
    ))}
