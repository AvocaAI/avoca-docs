---
title: 'Create AI Assistant'
api: 'POST /assistants'
description: 'Configure a new AI voice assistant for handling calls'
---

<Note>
  This endpoint creates and configures a new AI assistant that can handle inbound calls, make outbound calls, and interact with customers using natural language.
</Note>

## Request

<ParamField body="name" type="string" required>
  A descriptive name for your AI assistant (e.g., "Main Receptionist", "After Hours Assistant")
</ParamField>

<ParamField body="type" type="string" required>
  The type of assistant. Options:
  - `inbound` - Handles incoming calls
  - `outbound` - Makes outgoing calls
  - `both` - Handles both inbound and outbound
</ParamField>

<ParamField body="voice" type="string" default="emily">
  The voice to use for the assistant. Available voices:
  - `emily` - Professional female voice
  - `michael` - Professional male voice  
  - `sarah` - Friendly female voice
  - `james` - Friendly male voice
</ParamField>

<ParamField body="language" type="string" default="en-US">
  Language and locale for the assistant. Supported:
  - `en-US` - English (United States)
  - `en-GB` - English (United Kingdom)
  - `es-US` - Spanish (United States)
  - `es-MX` - Spanish (Mexico)
</ParamField>

<ParamField body="greeting" type="string" required>
  The initial greeting the assistant will use when answering calls
</ParamField>

<ParamField body="knowledge_base" type="object">
  Information about your business that the assistant can reference
  
  <Expandable title="properties">
    <ParamField body="services" type="array">
      List of services your business offers
    </ParamField>
    
    <ParamField body="service_area" type="array">
      Geographic areas you serve
    </ParamField>
    
    <ParamField body="business_hours" type="string">
      Your business hours in natural language
    </ParamField>
    
    <ParamField body="pricing" type="object">
      Pricing information the assistant can share
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="transfer_rules" type="object">
  Rules for when the assistant should transfer calls to humans
  
  <Expandable title="properties">
    <ParamField body="emergency" type="object">
      Emergency transfer configuration
      
      <Expandable title="properties">
        <ParamField body="keywords" type="array">
          Keywords that trigger emergency transfer
        </ParamField>
        
        <ParamField body="action" type="string">
          Action to take (e.g., "transfer_to_oncall")
        </ParamField>
      </Expandable>
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="booking_enabled" type="boolean" default="true">
  Whether the assistant can schedule appointments
</ParamField>

<ParamField body="servicetitan_integration" type="boolean" default="false">
  Enable ServiceTitan integration for automatic job creation
</ParamField>

## Response

<ResponseField name="id" type="string">
  Unique identifier for the assistant
</ResponseField>

<ResponseField name="created_at" type="string">
  ISO 8601 timestamp of when the assistant was created
</ResponseField>

<ResponseField name="phone_number" type="string">
  The phone number assigned to this assistant
</ResponseField>

<ResponseField name="status" type="string">
  Current status of the assistant:
  - `active` - Ready to handle calls
  - `provisioning` - Being set up
  - `paused` - Temporarily disabled
</ResponseField>
