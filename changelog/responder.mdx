import { updatesData } from '/snippets/updates.jsx';

## Recent Updates

{(() => {
  // Filter for past updates only (before today)
  const todayISO = new Date().toISOString().split('T')[0]; // Get today's date in YYYY-MM-DD format
  
  const pastUpdates = updatesData.updates
    .filter(update => update.product === "Responder")
    .filter(update => {
      return update.date < todayISO;
    })
    .sort((a, b) => b.date.localeCompare(a.date)); // Sort by date descending
  
  return pastUpdates.map((update, index) => (
    <Update key={index} label={update.displayDate}>
      <div style={{ marginBottom: '2rem' }}>
        <h3>{update.title}</h3>
        {update.author && <p><em>Author: {update.author}</em></p>}
        {update.description && <p>{update.description}</p>}
        {update.items && update.items.length > 0 && (
          <ul>
            {update.items.map((item, iIndex) => (
              <li key={iIndex}>
                <strong>{item.label}</strong> - {item.description}
              </li>
            ))}
          </ul>
        )}
        {update.note && <Note>{update.note}</Note>}
      </div>
    </Update>
  ));
})()}
